Replace `backend/prisma/schema.prisma` with the following complete schema:

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "postgresql"
  url      = env("DATABASE_URL")
}

// ─── Enums ────────────────────────────────────────────
// Mirror frontend type unions exactly

enum UserRole {
  user
  admin
}

enum LoanStatus {
  pending
  active
  overdue
  returned
  cancelled
}

enum NotificationType {
  approval
  reminder
  rejection
  new_item
  overdue
}

// ─── Models ───────────────────────────────────────────

/// User account — maps to frontend/types/user.ts User interface
/// and frontend/types/lab-session.ts LabSessionUser (with tag join)
model User {
  id        String   @id @default(cuid())
  name      String
  email     String   @unique
  password  String   // Plain text in Phase 5 seed, hashed via bcrypt in Phase 6
  matricula String?
  role      UserRole @default(user)
  phone     String?
  avatarUrl String?
  isActive  Boolean  @default(true)
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  // Tag-based permission system
  tag   Tag?    @relation(fields: [tagId], references: [id])
  tagId String?

  // Relations
  loans               Loan[]
  notifications       Notification[]
  permissionOverrides UserPermissionOverride[]

  @@map("users")
}

/// Tag/Role — maps to frontend/types/settings.ts Tag interface
/// permissions field stores TagPermissions as JSON (11 boolean keys)
model Tag {
  id          String   @id @default(cuid())
  name        String   @unique
  color       String   // Hex color like "#0f62fe" (frontend settings mock format)
  description String
  permissions Json     // { ver_itens: bool, pedir_emprestimos: bool, ... } — 11 keys
  createdAt   DateTime @default(now())
  updatedAt   DateTime @updatedAt

  // Relations
  users               User[]
  permissionOverrides UserPermissionOverride[]

  @@map("tags")
}

/// Inventory category — maps to frontend/types/settings.ts Category interface
/// linkedItemsCount is computed via _count.items, not stored
model Category {
  id        String   @id @default(cuid())
  name      String   @unique
  createdAt DateTime @default(now())
  updatedAt DateTime @updatedAt

  items InventoryItem[]

  @@map("categories")
}

/// Lab equipment item — maps to frontend/types/lab-inventory.ts LabInventoryListItem
/// Note: frontend stores category as string NAME, backend uses FK.
/// API response must include category.name as "category" string field.
model InventoryItem {
  id                String   @id @default(cuid())
  name              String
  description       String?
  quantity          Int      @default(0)
  availableQuantity Int      @default(0)
  loanedQuantity    Int      @default(0)
  image             String?  // URL or path to item image
  isActive          Boolean  @default(true)
  createdAt         DateTime @default(now())
  updatedAt         DateTime @updatedAt

  category   Category @relation(fields: [categoryId], references: [id])
  categoryId String

  loanItems LoanItem[]

  @@map("inventory_items")
}

/// Loan request — maps to frontend/types/loan.ts Loan interface
/// borrowerName and borrowerEmail are denormalized in API response
/// (fetched via borrower relation, not stored separately)
model Loan {
  id             String     @id @default(cuid())
  status         LoanStatus @default(pending)
  loanDate       DateTime
  dueDate        DateTime
  returnedDate   DateTime?
  notes          String?
  labObservation String?    // Admin observation when approving/rejecting
  returnedLate   Boolean    @default(false)
  createdAt      DateTime   @default(now())
  updatedAt      DateTime   @updatedAt

  borrower   User   @relation(fields: [borrowerId], references: [id])
  borrowerId String

  items LoanItem[]

  @@map("loans")
}

/// Individual item in a loan — maps to frontend/types/loan.ts LoanItem
/// inventoryItemName is denormalized for quick display without joining
model LoanItem {
  id                String @id @default(cuid())
  quantity          Int
  inventoryItemName String // Snapshot of item name at time of loan

  loan   Loan   @relation(fields: [loanId], references: [id], onDelete: Cascade)
  loanId String

  inventoryItem   InventoryItem @relation(fields: [inventoryItemId], references: [id])
  inventoryItemId String

  @@map("loan_items")
}

/// User notification — maps to frontend/mocks/notifications.ts LabNotification
model Notification {
  id        String           @id @default(cuid())
  title     String
  body      String
  read      Boolean          @default(false)
  type      NotificationType
  createdAt DateTime         @default(now())

  user   User   @relation(fields: [userId], references: [id], onDelete: Cascade)
  userId String

  @@map("notifications")
}

/// Per-user permission override — maps to frontend/types/settings.ts UserPermissionOverride
/// customOverrides stores Partial<TagPermissions> as JSON (only the overridden keys)
model UserPermissionOverride {
  id              String @id @default(cuid())
  customOverrides Json   // e.g. { "manipular_estoque": true }

  user   User   @relation(fields: [userId], references: [id], onDelete: Cascade)
  userId String

  tag   Tag    @relation(fields: [tagId], references: [id])
  tagId String

  @@unique([userId, tagId]) // One override per user-tag combination
  @@map("user_permission_overrides")
}
```

import { PrismaClient } from '@prisma/client';

/**
 * Prisma Client singleton.
 * Uses globalThis to prevent multiple instances during hot-reload in development.
 */
const globalForPrisma = globalThis as unknown as {
  prisma: PrismaClient | undefined;
};

export const prisma =
  globalForPrisma.prisma ??
  new PrismaClient({
    log:
      process.env.NODE_ENV === 'development'
        ? ['warn', 'error']
        : ['error'],
  });

if (process.env.NODE_ENV !== 'production') {
  globalForPrisma.prisma = prisma;
}

Replace `backend/src/routes/health.ts`:

```typescript
import { Router } from 'express';
import { prisma } from '../config/database';
import { sendSuccess } from '../utils/response';

const router = Router();

/**
 * GET /api/health
 * Returns server health status with database connectivity check.
 * Response: ApiResponse<{ status: "ok" | "degraded", database: "connected" | "disconnected" }>
 */
router.get('/', async (_req, res) => {
  try {
    await prisma.$queryRaw`SELECT 1`;
    sendSuccess(
      res,
      { status: 'ok', database: 'connected' },
      'Server is healthy'
    );
  } catch {
    sendSuccess(
      res,
      { status: 'degraded', database: 'disconnected' },
      'Server running but database unavailable'
    );
  }
});

export default router;
```

const server = app.listen(env.PORT, () => {

  console.log(`   URL:         http://localhost:${env.PORT}`);
  console.log(`   Environment: ${env.NODE_ENV}`);
  console.log(`   CORS origin: ${env.CORS_ORIGIN}`);
  console.log('');
});

// ─── Graceful shutdown ────────────────────────
async function shutdown(signal: string): Promise<void> {
  console.log(`\n${signal} received. Shutting down gracefully...`);
  server.close(async () => {
    await prisma.$disconnect();
    console.log('Server and database connections closed.');
    process.exit(0);
  });
}

process.on('SIGTERM', () => shutdown('SIGTERM'));
process.on('SIGINT', () => shutdown('SIGINT'));
